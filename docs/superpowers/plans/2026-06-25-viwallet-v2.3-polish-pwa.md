**Goal:** Final polish (micro-interactions, empty states, haptics), a performance pass, and PWA hardening (icons, install prompt, offline fallback, update toast) — no new features.

**Architecture:** Reuse v2.1 motion primitives; add a small set of polish components (`SuccessCheck`, `Confetti`, `EmptyState`, haptics util) and PWA infrastructure (enhanced `manifest.json`, extended `public/sw.js`, install + update components). - [ ] **Step 3: Implement haptics**

`lib/haptics.ts`:
```ts
const KEY = 'viwallet:haptics'

export function shouldVibrate(supported: boolean, enabled: boolean): boolean {
  return supported && enabled
}

export function hapticsEnabled(): boolean {
  if (typeof window === 'undefined') return true
  return window.localStorage.getItem(KEY) !== 'off'
}

export function setHapticsEnabled(on: boolean): void {
  if (typeof window !== 'undefined') window.localStorage.setItem(KEY, on ? 'on' : 'off')
}

export function haptic(pattern: number | number[] = 10): void {
  const supported = typeof navigator !== 'undefined' && 'vibrate' in navigator
  if (shouldVibrate(supported, hapticsEnabled())) navigator.vibrate(pattern)
}
```

- [ ] **Step 3: Implement version compare**

`lib/sw-version.ts`:
```ts
export function isNewerVersion(current: string, incoming: string): boolean {
  const a = current.split('.').map(Number)
  const b = incoming.split('.').map(Number)
  for (let i = 0; i < 3; i++) {
    if ((b[i] ?? 0) > (a[i] ?? 0)) return true
    if ((b[i] ?? 0) < (a[i] ?? 0)) return false
  }
  return false
}
```
